// Valid Parentheses
// Difficulty: Easy
// https://leetcode.com/problems/valid-parentheses/

bool isValid(char* s)
{
    int n = strlen(s);
    char *stack = malloc(n);
    int top = -1;
    for (int i=0; s[i]!='\0'; i++){
        if(s[i]=='('){
            stack[++top] = ')';
        }
        else if(s[i] == '{'){
            stack[++top] = '}';
        }
        else if(s[i] == '['){
            stack[++top] = ']';
        }
        else {
            if (top==-1 || stack[top--] != s[i]){
                free(stack);
                return false;
            }
        }
    }
    bool result = (top ==-1);
    free(stack);
    return result;
}
